# Function rotate to orient the data properly vertically and horizontally
HEIGHT = 28
WIDTH  = 28

def rotate(image):
    image = image.reshape([HEIGHT, WIDTH])
    image = np.fliplr(image)
    image = np.rot90(image)
    return image

model.add(Flatten())

model.add(Dropout(.5))
model.add(Dense(units=num_classes, activation='softmax'))
